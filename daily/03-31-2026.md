    HouseState, COUNT(*)
FROM

GROUP BY HouseState;

```mysql
SELECT
    HouseState, HouseBuildYear, COUNT(*)
FROM
    HOUSE
GROUP BY HouseBuildYear, HouseState WITH ROLLUP;
```

```mysql
SELECT
    ChannelID AS Channel,
    MONTH(StayStartDate) AS Month,
    COUNT(*) AS Total
FROM
    STAY
GROUP BY ChannelID , MONTH(StayStartDate) WITH ROLLUP;
```


```mysql
SELECT
    HouseID, MONTH(StayStartDate), COUNT(*)
FROM
    STAY
GROUP BY HouseID, MONTH(StayStartDate) WITH ROLLUP
ORDER BY HouseID, MONTH(StayStartDate);
```


```mysql
SELECT
    GuestID, MONTH(StayStartDate), COUNT(*)
FROM
    STAY
WHERE
    MONTH(StayStartDate ) > 6
GROUP BY GuestID, MONTH(StayStartDate) WITH ROLLUP;
```
