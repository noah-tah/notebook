# The AI slop in question

```

# ---- ssh-agent shared across terminals ----
export SSH_ENV="$HOME/.ssh/agent_env"

start_agent() {
  umask 077
  ssh-agent -s > "$SSH_ENV"
  . "$SSH_ENV" >/dev/null
}

# Load existing agent env if present
if [ -f "$SSH_ENV" ]; then
  . "$SSH_ENV" >/dev/null
fi

# Start new agent only if missing/dead
if [ -z "$SSH_AUTH_SOCK" ] || ! kill -0 "$SSH_AGENT_PID" 2>/dev/null; then
  start_agent
fi

# Add key only if not already loaded
ssh-add -l >/dev/null 2>&1 || ssh-add ~/.ssh/id_ed25519

```
